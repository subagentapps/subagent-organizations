# Step 4 — Per-plugin pages, ADR rendering, changelog route

Adds four new routes to the site. Keeps the same flat-dark aesthetic and teal
accent. No new dependencies except a markdown renderer.

New routes:

| Path | Purpose |
|---|---|
| `/` | Dashboard (existing) |
| `/plugins/:name` | Per-plugin landing page, templated from GitHub |
| `/adr` + `/adr/:number` | ADR index + individual ADR view |
| `/changelog` | Combined changelog across the 14 plugins |

---

## 1. Add a router

Use `wouter` — 2kb, hook-based, composable. Install:

```bash
npm install wouter
npm install -D @types/unist  # for markdown AST types later
```

Also add markdown rendering:

```bash
npm install marked dompurify
npm install -D @types/marked @types/dompurify
```

`marked` is 30kb, zero deps. `DOMPurify` sanitizes HTML post-render.

## 3. Pages Function: `/api/github-file`

Shared function to fetch arbitrary Markdown from the kwpc repo.
Create `functions/api/github-file.ts`:

```typescript
interface Env {
  GITHUB_TOKEN: string;
  AKW_CACHE: KVNamespace;
}

const CACHE_TTL = 300;

export const onRequestGet: PagesFunction<Env> = async ({ request, env }) => {
  const url = new URL(request.url);
  const path = url.searchParams.get('path');
  const repo = url.searchParams.get('repo') || 'knowledge-work-plugins-cli';
  const ref = url.searchParams.get('ref') || 'main';

  if (!path) return new Response('missing path', { status: 400 });

  // Security: only allow docs/, ADR, CHANGELOG, README paths
  const allowed = /^(docs\/|CHANGELOG\.md$|README\.md$|[a-z0-9-]+\/README\.md$|[a-z0-9-]+\/CHANGELOG\.md$)/i;
  if (!allowed.test(path)) return new Response('path not allowed', { status: 403 });

  const cacheKey = `file:${repo}:${ref}:${path}`;
  const cached = await env.AKW_CACHE.get(cacheKey);
  if (cached) {
    return new Response(cached, {
      headers: { 'content-type': 'text/markdown', 'x-akw-cache': 'hit' },
    });
  }

  const ghUrl = `https://api.github.com/repos/agentknowledgeworkers/${repo}/contents/${path}?ref=${ref}`;
  const res = await fetch(ghUrl, {
    headers: {
      'authorization': `bearer ${env.GITHUB_TOKEN}`,
      'accept': 'application/vnd.github.raw',
      'user-agent': 'agentknowledgeworkers-tracker',
    },
  });

  if (!res.ok) return new Response(`github: ${res.status}`, { status: res.status });

  const body = await res.text();
  await env.AKW_CACHE.put(cacheKey, body, { expirationTtl: CACHE_TTL });

  return new Response(body, {
    headers: { 'content-type': 'text/markdown', 'x-akw-cache': 'miss' },
  });
};
```

## 4. Markdown renderer hook

Create `src/lib/useMarkdown.ts`:

```typescript
import { useEffect, useState } from 'react';
import { marked } from 'marked';
import DOMPurify from 'dompurify';

marked.setOptions({
  gfm: true,
  breaks: false,
});

export function useMarkdown(path: string, repo = 'knowledge-work-plugins-cli') {
  const [html, setHtml] = useState<string>('');
  const [raw, setRaw] = useState<string>('');
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  useEffect(() => {
    let cancelled = false;
    setLoading(true);
    setError(null);

    const url = `/api/github-file?path=${encodeURIComponent(path)}&repo=${encodeURIComponent(repo)}`;
    fetch(url)
      .then((r) => r.ok ? r.text() : Promise.reject(r.statusText))
      .then(async (md) => {
        if (cancelled) return;
        setRaw(md);
        const rendered = await marked.parse(md);
        setHtml(DOMPurify.sanitize(rendered as string));
      })
      .catch((e) => !cancelled && setError(String(e)))
      .finally(() => !cancelled && setLoading(false));

    return () => { cancelled = true; };
  }, [path, repo]);

  return { html, raw, loading, error };
}
```

```typescript
import { useParams } from 'wouter';
import PageShell from './_PageShell';
import { useMarkdown } from '@/lib/useMarkdown';
import { useProjectDetail } from '@/lib/useProjects';
import MarkdownBody from '@/components/MarkdownBody';

const C = { bg: '#000000', surface: '#0A0A0A', border: '#1A1A1A',
            text: '#F5F2ED', textDim: '#8F9997', textMute: '#5A6462',
            teal: '#0097A7', tealHi: '#00BCD4' };
const FONT = { sans: `'Inter', sans-serif`, mono: `'JetBrains Mono', monospace` };

export default function PluginPage() {
  const params = useParams<{ name: string }>();
  const name = params.name;
  const { html, loading, error } = useMarkdown(`${name}/README.md`);
  const { detail } = useProjectDetail(name);

  return (
    <PageShell>
      <div style={{ maxWidth: 920, margin: '0 auto', padding: '40px 24px 80px' }}>
        {/* Eyebrow + title */}
        <div style={{
          fontFamily: FONT.mono, fontSize: 11, color: C.textMute,
          letterSpacing: '0.12em', textTransform: 'uppercase', marginBottom: 8,
        }}>
          Plugin · {inferTier(name)}
        </div>
        <h1 style={{
          fontFamily: FONT.sans, fontSize: 32, fontWeight: 500,
          color: C.text, margin: 0, letterSpacing: '-0.015em',
        }}>
          {name}
        </h1>

        {/* Status strip */}
        {detail && (
          <div style={{
            marginTop: 16, display: 'flex', gap: 24,
            padding: '16px 0', borderTop: `1px solid ${C.border}`, borderBottom: `1px solid ${C.border}`,
          }}>
            <StatItem label="Milestones" value={detail.milestones.length} />
            <StatItem label="Issues" value={detail.milestones.reduce((a, m) => a + m.issues.length, 0)} />
            <StatItem label="Tests passing" value={countPassingTests(detail) + ' / ' + countTotalTests(detail)} />
          </div>
        )}

        {/* README body */}
        <div style={{ marginTop: 32 }}>
          {loading && <div style={{ color: C.textMute, fontFamily: FONT.mono, fontSize: 13 }}>Loading README from GitHub…</div>}
          {error && <div style={{ color: '#EF4444', fontFamily: FONT.mono, fontSize: 13 }}>Failed to load: {error}</div>}
          {!loading && !error && <MarkdownBody html={html} />}
        </div>
      </div>
    </PageShell>
  );
}

function StatItem({ label, value }: { label: string; value: number | string }) {
  return (
    <div>
      <div style={{ fontFamily: FONT.mono, fontSize: 10, color: C.textMute, letterSpacing: '0.08em', textTransform: 'uppercase' }}>{label}</div>
      <div style={{ fontFamily: FONT.mono, fontSize: 18, color: C.text, marginTop: 4 }}>{value}</div>
    </div>
  );
}

function inferTier(name: string): string {
  const T1 = ['legal-cli', 'finance-cli', 'marketing-cli', 'operations-cli', 'human-resources-cli', 'customer-support-cli', 'sales-cli'];
  const T3 = ['pdf-viewer-cli'];
  if (T1.includes(name)) return 'T1';
  if (T3.includes(name)) return 'T3';
  return 'T2';
}

function countPassingTests(detail: any): number {
  return detail.milestones.reduce((acc: number, m: any) =>
    acc + m.issues.filter((i: any) => i.state === 'closed' && i.title.match(/^T\d: /)).length, 0);
}

function countTotalTests(detail: any): number {
  return detail.milestones.reduce((acc: number, m: any) =>
    acc + m.issues.filter((i: any) => i.title.match(/^T\d: /)).length, 0);
}
```

And the Pages Function `functions/api/releases.ts`:

```typescript
interface Env { GITHUB_TOKEN: string; AKW_CACHE: KVNamespace; }

export const onRequestGet: PagesFunction<Env> = async ({ env }) => {
  const cacheKey = 'releases:all';
  const cached = await env.AKW_CACHE.get(cacheKey, 'json');
  if (cached) return Response.json(cached, { headers: { 'x-akw-cache': 'hit' } });

  const data = await fetch(
    'https://api.github.com/graphql',
    {
      method: 'POST',
      headers: {
        'authorization': `bearer ${env.GITHUB_TOKEN}`,
        'content-type': 'application/json',
        'user-agent': 'agentknowledgeworkers-tracker',
      },
      body: JSON.stringify({
        query: `{
          repository(owner: "agentknowledgeworkers", name: "knowledge-work-plugins-cli") {
            releases(first: 50, orderBy: {field: CREATED_AT, direction: DESC}) {
              nodes { tagName name publishedAt description url }
            }
          }
        }`,
      }),
    },
  ).then((r) => r.json());

  const releases = (data.data?.repository?.releases?.nodes || []).map((r: any) => ({
    tagName: r.tagName,
    name: r.name,
    publishedAt: r.publishedAt,
    body: r.description || '',
    url: r.url,
    plugin: (r.tagName.split('-v')[0] || 'unknown'),
  }));

  const payload = { releases, fetchedAt: new Date().toISOString() };
  await env.AKW_CACHE.put(cacheKey, JSON.stringify(payload), { expirationTtl: 600 });

  return Response.json(payload, { headers: { 'x-akw-cache': 'miss' } });
};
```
